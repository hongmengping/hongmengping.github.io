对象解构

什么是解构

解构不会被解构的目标造成任何影响

const user = {
    name: 'kevin',
    age: 11,
    sex: '男',
    address: {
        province: '四川',
        city: '成都'
    }
}

// 原来的写法
let name, age, sex, address;
name = user.name;
age = user.age;
sex = user.sex;
address = user.address;

// ES6解构
let name, age, sex, address;
({name, age = 12, sex, address} = user); //要用小括号 让他们变为一个整体

// 合并书写  先定义4个变量，然后从对象中读取同名属性，放到变量中
let {name, age, sex, address} = user;

使用ES6的语法规则，将一个对象或数组的某个属性提取到某个变量中

在解构中使用默认值

{同名变量 = 默认值}

非同名属性解构


{属性名: 变量名}

const user = {
    name: 'kevin',
    age: 11,
    sex: '男',
    address: {
        province: '四川',
        city: '成都'
    }
}

// 解构出name，然后剩余的所有属性，放到一个新的对象中，变量名为obj
// name: kevin
// obj: {age: 11, sex: "男", address: {...}}
const {name, ...obj} = user;


// 先定义4个变量：name、age、gender、address
// 再从对象user中读取同名属性赋值（其中gender读取的是sex属性）
let { name, age, sex, address } = user;

// 解构的比较深  解构出user中的name、province  
// 这里的address不会当成是变量 会对address进行进一步解构
// 定义两个变量name、province
// 再解构
const { name, address: { province } } = user;

console.log(name, address, province) // 会报错 见下图

数组解构

// 单纯数组
const numbers = ["a", "b", "c", "d"];

const {
    0: n1,
    1: n2
} = numbers;

//更加清新脱俗的写法
const [n1, n2] = numbers;
const [n1, , , n4, n5] = numbers;  //a d undefined

console.log(n1, n2);

// 数组包含数组的解构
const numbers2 = ["a", "b", "c", "d", [1, 2, 3, 4]];

//得到numbers下标为4的数组汇总的下标为2的数据，放到变量中
const [, , , , [, , n]] = numbers

console.log(n) // 3

// 数组包含对象的解构
const numbers2 = ["a", "b", "c", "d", {
    a: 1,
    b: 2
};

// 得到numbers下标为4的数组属性a，赋值给变量a
const [, , , , { a }] = numbers;

console.log(a)

// 换一种写法
const {a: A} = numbers[4];

const nubmers = [345, 7, 23, 5, 234];

// 得到数组前两项，分别放到变量a和b中，然后剩余的所有数据放到数组nums
const [a, b, ...nums] = numbers;

console.log(a, b, nums);

// 交换变量
let a = 1, b = 2;

[b, a] = [a, b];

console.log(a, b); //2, 1

// 小例子
const article = {
    title: "文章标题",
    content: "文章内容",
    comments: [{
        content: "评论1",
        user: {
            id: 1,
            name: '用户名1'
        }
    }, {
        content: "评论2",
        user: {
            id: 2, 
            name: "用户名2"
        }
    }]
}

// 解构出第二条评论的用户名和评论内容
// name: "用户名2" content: "评论2"

const {
    comments: [, {
        content,
        user: {
            name
        }
    }]
} = article;

console.log(content, name) // 评论2 用户名2

const {
    content,
    user: {
        name
    }
} = article.comments[1];

console.log(content, name)

参数解构

// es5的写法
function print(user) {
    console.log(`姓名：${user.name}`);
    console.log(`年龄：${user.age}`);
    console.log(`性别：${user.sex}`);
    console.log(`身份：${user.address.province}`);
    console.log(`城市：${user.address.city}`);
}

// es6的写法 用解构的方式来写
function print({name, age, sex, address: {
    province,
    city
}}) {
    console.log(`姓名：${name}`);
    console.log(`年龄：${age}`);
    console.log(`性别：${sex}`);
    console.log(`身份：${province}`);
    console.log(`城市：${city}`);
}

const user = {
    name: "kevin",
    age: 11,
    sex: "男",
    address: {
        province: "四川",
        city: "成都"
    }
}

print(user)
// ajax中运用 旧的写法
function ajax(options) {
    const defaultOptions = {
        method: "get",
        url: "/"
    }
    const opt = {
        ...defaultOptions,
        ...options
    }
    console.log(opt)
}

// ajax中 运用解构的写法 把参数解构出来
// {method = "get" ,url = "/"} = {}  给解构附上默认值 如果没传也不会报错
function ajax({
    method = "get",
    url = "/"
} = {}) {
    console.log(method, url)
}

ajax({
    url: "/abc"
})
